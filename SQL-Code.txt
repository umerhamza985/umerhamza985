CREATE TABLE PATIENT
(PATIENT_ID INT PRIMARY KEY,
NAME VARCHAR(30),
GENDER VARCHAR(6),
AGE INT,
ADDRESS VARCHAR(100)
);

CREATE TABLE PATIENT_DETAILS
(PATIENT_ID INT,
BLOOD_GROUP VARCHAR(10),
LAST_VISIT DATE,
MEDICATIONS VARCHAR(200),
FOREIGN KEY (PATIENT_ID) REFERENCES PATIENT(PATIENT_ID));

CREATE TABLE PATIENT_CONDITION 
(PATIENT_ID INT,
BLOOD_PRESSURE INT,
TEMPERATURE INT,
TESTS_CONDUCTED VARCHAR(200),
OXYGEN_LEVEL INT,
HEART_RATE INT,
FOREIGN KEY (PATIENT_ID) REFERENCES PATIENT(PATIENT_ID)
);

CREATE TABLE PATIENT_ISSUE
(PATIENT_ID INT,
CURRENT_DISEASE VARCHAR(50),
DETAILS VARCHAR(200),
DATE_NOW DATE,
FOREIGN KEY (PATIENT_ID) REFERENCES PATIENT(PATIENT_ID)
);

CREATE TABLE PATIENT_UNDER_OBSERVATION
(PATIENT_ID INT,
ROOM_NO INT,
CURRENT_CONDITION VARCHAR(20),
DISEASE VARCHAR(50),
NEXT_OBSERVATION DATE,
FOREIGN KEY (PATIENT_ID) REFERENCES PATIENT(PATIENT_ID));

CREATE TABLE STAFF
(STAFF_ID INT PRIMARY KEY,
FIRST_NAME VARCHAR(50),
LAST_NAME VARCHAR(50),
DATE_OF_BIRTH DATE,
EMAIL VARCHAR(100),
PHONE VARCHAR(16),
ADDRESS VARCHAR(100),
PICTURE BLOB
);


CREATE TABLE STAFF_LOGIN 
(STAFF_ID INT,
USERNAME VARCHAR(100),
PASSWORD VARCHAR(50),
FOREIGN KEY (STAFF_ID) REFERENCES STAFF(STAFF_ID)
);
